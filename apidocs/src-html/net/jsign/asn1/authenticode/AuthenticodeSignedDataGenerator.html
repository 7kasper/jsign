017    package net.jsign.asn1.authenticode;
018    
019    import java.io.IOException;
020    import java.io.OutputStream;
021    
022    import org.bouncycastle.asn1.ASN1Encodable;
023    import org.bouncycastle.asn1.ASN1ObjectIdentifier;
024    import org.bouncycastle.asn1.ASN1Set;
025    import org.bouncycastle.asn1.DERSet;
026    import org.bouncycastle.asn1.cms.CMSObjectIdentifiers;
027    import org.bouncycastle.asn1.cms.ContentInfo;
028    import org.bouncycastle.asn1.cms.SignerInfo;
029    import org.bouncycastle.cms.CMSException;
030    import org.bouncycastle.cms.CMSProcessableByteArray;
031    import org.bouncycastle.cms.CMSSignedData;
032    import org.bouncycastle.cms.CMSSignedDataGenerator;
033    import org.bouncycastle.cms.SignerInfoGenerator;
034    import org.bouncycastle.cms.SignerInformation;
035    
036    /**
037     * Simplified SignedDataGenerator suitable for Authenticode signing.
038     * 
039     * @author Emmanuel Bourg
040     * @since 1.0
041     */
042    public class AuthenticodeSignedDataGenerator extends CMSSignedDataGenerator {
043        
044        public CMSSignedData generate(ASN1ObjectIdentifier contentTypeOID, ASN1Encodable content) throws CMSException, IOException {
045            digests.clear();
046            
047            SignerInfo signerInfo;
048            
049            if (!_signers.isEmpty()) {
050                signerInfo = ((SignerInformation) _signers.get(0)).toASN1Structure();
051            } else {
052                SignerInfoGenerator signerInfoGenerator = (SignerInfoGenerator) signerGens.get(0);
053    
054                try {
055                    byte[] signedContent = content.toASN1Primitive().getEncoded("DER");
056                    
057                    OutputStream out = signerInfoGenerator.getCalculatingOutputStream();
058                    out.write(signedContent, 2, signedContent.length - 2); // skip the first 2 bytes as specified
059                    out.flush();
060                    out.close();
061                } catch (IOException e) {
062                    throw new CMSException("data processing exception: " + e.getMessage(), e);
063                }
064                
065                signerInfo = signerInfoGenerator.generate(contentTypeOID);
066                
067                byte[] calculatedDigest = signerInfoGenerator.getCalculatedDigest();
068                digests.put(signerInfoGenerator.getDigestAlgorithm().getAlgorithm().getId(), calculatedDigest);
069            }       
070    
071            ContentInfo encInfo = new ContentInfo(contentTypeOID, content);
072            ASN1Set certificates = new DERSet((ASN1Encodable[]) certs.toArray(new ASN1Encodable[0]));
073    
074            ASN1Encodable signedData = new AuthenticodeSignedData(signerInfo.getDigestAlgorithm(), encInfo, certificates, signerInfo);
075    
076            ContentInfo contentInfo = new ContentInfo(CMSObjectIdentifiers.signedData, signedData);
077    
078            return new CMSSignedData(new CMSProcessableByteArray(contentTypeOID, content.toASN1Primitive().getEncoded("DER")), contentInfo);
079        }
080    }
