078    /**
079     * Sign a portable executable file. Timestamping is enabled by default
080     * and relies on the Comodo server (http://timestamp.comodoca.com/authenticode).
081     * 
082     * @see <a href="http://download.microsoft.com/download/9/c/5/9c5b2167-8017-4bae-9fde-d599bac8184a/Authenticode_PE.docx">Windows Authenticode Portable Executable Signature Format</a>
083     * @see <a href="http://msdn.microsoft.com/en-us/library/windows/desktop/bb931395%28v=vs.85%29.aspx?ppud=4">Time Stamping Authenticode Signatures</a>
084     * 
085     * @author Emmanuel Bourg
086     * @since 1.0
087     */
088    public class PESigner {

140        /**
141         * Sign the specified executable file.
142         * @throws Exception
143         */
144        public void sign(PEFile file) throws Exception {
145            // pad the file on a 8 byte boundary
146            // todo only if there was no previous certificate table
147            file.pad(8);
148            
149            // compute the signature
150            byte[] certificateTable = createCertificateTable(file);
151            
152            file.writeDataDirectory(DataDirectoryType.CERTIFICATE_TABLE, certificateTable);
153            file.close();
154        }
155    
156        private byte[] createCertificateTable(PEFile file) throws IOException, CMSException, OperatorCreationException, CertificateEncodingException {
157            CMSSignedData sigData = createSignature(file);
158            
159            if (timestamping) {
160                sigData = timestamp(sigData);
161            }
162            
163            // pad the table
164            byte[] signature = sigData.toASN1Structure().getEncoded("DER");
165            signature = pad(signature, 8);
166            
167            // add the header
168            ByteBuffer buffer = ByteBuffer.allocate(signature.length + 8);
169            buffer.order(ByteOrder.LITTLE_ENDIAN);
170            buffer.putInt(buffer.limit());
171            buffer.putShort((short) 0x0200);
172            buffer.putShort((short) 0x0002);
173            buffer.put(signature);
174            
175            return buffer.array();
176        }
177    
178        private byte[] pad(byte[] data, int multiple) {
179            if (data.length % multiple == 0) {
180                return data;
181            } else {
182                byte[] copy = new byte[data.length + (multiple - data.length % multiple)];
183                System.arraycopy(data, 0, copy, 0, data.length);
184                return copy;
185            }
186        }
187    
188        private CMSSignedData createSignature(PEFile file) throws IOException, CMSException, OperatorCreationException, CertificateEncodingException {
189            byte[] sha1 = file.computeDigest("SHA1");
190            
191            AlgorithmIdentifier algorithmIdentifier = new AlgorithmIdentifier(X509ObjectIdentifiers.id_SHA1, DERNull.INSTANCE);
192            DigestInfo digestInfo = new DigestInfo(algorithmIdentifier, sha1);
193            SpcIndirectDataContent spcIndirectDataContent = new SpcIndirectDataContent(digestInfo);
194            
195            ContentSigner sha1Signer = new JcaContentSignerBuilder("SHA1with" + privateKey.getAlgorithm()).build(privateKey);
196            DigestCalculatorProvider digestCalculatorProvider = new JcaDigestCalculatorProviderBuilder().build();
197            
198            // prepare the authenticated attributes
199            CMSAttributeTableGenerator attributeTableGenerator = new DefaultAuthenticatedAttributeTableGenerator(createAuthenticatedAttributes());
200            
201            // fetch the signing certificate
202            X509CertificateHolder certificate = new JcaX509CertificateHolder((X509Certificate) chain[0]);
203            
204            // prepare the signerInfo with the extra authenticated attributes
205            SignerInfoGeneratorBuilder signerInfoGeneratorBuilder = new SignerInfoGeneratorBuilder(digestCalculatorProvider);
206            signerInfoGeneratorBuilder.setSignedAttributeGenerator(attributeTableGenerator);
207            SignerInfoGenerator signerInfoGenerator = signerInfoGeneratorBuilder.build(sha1Signer, certificate);
208            
209            AuthenticodeSignedDataGenerator generator = new AuthenticodeSignedDataGenerator();
210            generator.addCertificates(new JcaCertStore(removeRoot(chain)));
211            generator.addSignerInfoGenerator(signerInfoGenerator);
212            
213            return generator.generate(AuthenticodeObjectIdentifiers.SPC_INDIRECT_DATA_OBJID, spcIndirectDataContent);
214        }

216        /**
217         * Remove the root certificate from the chain, unless the chain consists in a single self signed certificate.
218         */
219        private List<Certificate> removeRoot(Certificate[] certificates) {
220            List<Certificate> list = new ArrayList<Certificate>();
221            
222            if (certificates.length == 1) {
223                list.add(certificates[0]);
224            } else {
225                for (Certificate certificate : certificates) {
226                    if (!isSelfSigned((X509Certificate) certificate)) {
227                        list.add(certificate);
228                    }
229                }
230            }
231            
232            return list;
233        }
234    
235        private boolean isSelfSigned(X509Certificate certificate) {
236            return certificate.getSubjectDN().equals(certificate.getIssuerDN());
237        }
238    
239        /**
240         * Creates the authenticated attributes for the SignerInfo section of the signature.
241         */
242        private AttributeTable createAuthenticatedAttributes() {
243            List<Attribute> attributes = new ArrayList<Attribute>();
244            
245            SpcStatementType spcStatementType = new SpcStatementType(AuthenticodeObjectIdentifiers.SPC_INDIVIDUAL_SP_KEY_PURPOSE_OBJID);
246            attributes.add(new Attribute(AuthenticodeObjectIdentifiers.SPC_STATEMENT_TYPE_OBJID, new DERSet(spcStatementType)));
247            
248            if (programName != null || programURL != null) {
249                SpcSpOpusInfo spcSpOpusInfo = new SpcSpOpusInfo(programName, programURL);
250                attributes.add(new Attribute(AuthenticodeObjectIdentifiers.SPC_SP_OPUS_INFO_OBJID, new DERSet(spcSpOpusInfo)));
251            }
252            
253            return new AttributeTable(new DERSet(attributes.toArray(new ASN1Encodable[attributes.size()])));
254        }

256        /**
257         * Add a timestamp to the signature.
258         */
259        private CMSSignedData timestamp(CMSSignedData sigData) throws IOException, CMSException {
260            SignerInformation signerInformation = ((SignerInformation) sigData.getSignerInfos().getSigners().iterator().next());
261    
262            CMSSignedData token = timestamp(signerInformation.toASN1Structure().getEncryptedDigest().getOctets(), new URL(tsaurl));
263    
264            SignerInformation timestampSignerInformation = (SignerInformation) token.getSignerInfos().getSigners().iterator().next();
265    
266            Attribute counterSignature = new Attribute(CMSAttributes.counterSignature, new DERSet(timestampSignerInformation.toASN1Structure()));
267            
268            signerInformation = SignerInformation.replaceUnsignedAttributes(signerInformation, new AttributeTable(new DERSet(counterSignature)));
269            
270            // add the certificates for the timestamp authority
271            Collection<?> certificates = new ArrayList();
272            certificates.addAll(sigData.getCertificates().getMatches(null));
273            certificates.addAll(token.getCertificates().getMatches(null));
274            Store certificateStore = new CollectionStore(certificates);
275            
276            AuthenticodeSignedDataGenerator generator = new AuthenticodeSignedDataGenerator();
277            generator.addCertificates(certificateStore);
278            generator.addSigners(new SignerInformationStore(Arrays.asList(signerInformation)));
279            
280            ASN1ObjectIdentifier contentType = new ASN1ObjectIdentifier(sigData.getSignedContentTypeOID());
281            ASN1Encodable content = ASN1Sequence.getInstance(sigData.getSignedContent().getContent());
282                    
283            return generator.generate(contentType, content);
284        }

286        private CMSSignedData timestamp(byte[] encryptedDigest, URL tsaurl) throws IOException, CMSException {
287            AuthenticodeTimeStampRequest timestampRequest = new AuthenticodeTimeStampRequest(encryptedDigest);
288            
289            byte[] request = Base64.encodeBase64(timestampRequest.getEncoded("DER"), true);        
290            
291            HttpURLConnection conn = (HttpURLConnection) tsaurl.openConnection();
292            
293            conn.setDoOutput(true);
294            conn.setDoInput(true);
295            conn.setUseCaches(false);
296            conn.setRequestMethod("POST");
297            conn.setRequestProperty("Content-type", "application/octet-stream");
298            conn.setRequestProperty("Content-length", String.valueOf(request.length));
299            conn.setRequestProperty("Accept", "application/octet-stream");
300            conn.setRequestProperty("User-Agent", "Transport");
301            
302            conn.getOutputStream().write(request);
303            conn.getOutputStream().flush();
304            
305            if (conn.getResponseCode() >= 400) {
306                throw new IOException("Unable to complete the timestamping due to HTTP error: " + conn.getResponseCode() + " - " + conn.getResponseMessage());
307            }
308            
309            InputStream in = conn.getInputStream();
310            ByteArrayOutputStream bout = new ByteArrayOutputStream();
311            
312            byte[] buffer = new byte[4096];
313            int n;
314            while ((n = in.read(buffer)) != -1) {
315                bout.write(buffer, 0, n);
316            }
317            
318            byte[] response = Base64.decodeBase64(bout.toByteArray());
319            
320            return new CMSSignedData(response);
321        }
